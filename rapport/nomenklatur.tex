\section{Nomenklatur}\label{nomenklatur}
\makenomenclature
\renewcommand{\nomname}{}

\nomenclature{$K$}{Energien til et roterende legeme\nomunit{$\frac{1}{2}\cdot\vec{L}\cdot\vec{\omega}$}}
\nomenclature{$X$}{Matrise som beskriver rotasjonen til t-nøkkelen\nomunit{$\begin{bmatrix}x_{11}&x_{12}&x_{13}\\x_{21}&x_{22}&x_{23}\\x_{31}&x_{32}&x_{33}\end{bmatrix}$}}
\nomenclature{$\vec{L_b}$}{Dreiemoment\nomunit{$\int\int\int_M{r_b^2\vec{\omega}-\vec{r_b}(\vec{r_b}\cdot\vec{\omega})}\,dm$}}
\nomenclature{$\Omega$}{Matrise med komponentene til $\vec{\omega_b}$\nomunit{$\begin{bmatrix}0&-\omega_z&\omega_y\\\omega_z&0&-\omega_x\\-\omega_y&\omega_x&0\end{bmatrix}$}}
\nomenclature{$\vec{\omega_b}$}{Vinkelfarten\nomunit{$\begin{bmatrix}\omega_x&\omega_y&\omega_y\end{bmatrix}$}}
\nomenclature{$M$}{Massen til t-nøkkel}\nomunit{$12\pi\rho$}
\nomenclature{$\rho$}{Massetettheten til t-nøkkel\nomunit{$6.7\text{ gram}$}}
\nomenclature{$L_1$}{Lengde til håndtak\nomunit{$8\text{ cm}$}}
\nomenclature{$L_2$}{Lengde til sylinder\nomunit{$4\text{ cm}$}}
\nomenclature{$R_1$}{Radius til håndtak\nomunit{$1\text{ cm}$}}
\nomenclature{$R_2$}{Radius til sylinder\nomunit{$1\text{ cm}$}}
\nomenclature{$W$}{Tilnærming av X\nomunit{$W\approx X$}}
\nomenclature{$I$}{Treghetsmoment\nomunit{$\begin{bmatrix}I_{xx}&0&0\\0&I_{yy}&0\\0&0&I_{zz}\end{bmatrix}$}}
\nomenclature{$Id_{3\times3}$}{Identitetsmatrise\nomunit{$\begin{bmatrix}1&0&0\\0&1&0\\0&0&1\end{bmatrix}$}}
\printnomenclature
